JavaScript BOM (نموذج كائن المتصفح)

ما هو BOM؟

BOM (Browser Object Model) هو واجهة تسمح لـ JavaScript بالتفاعل مع المتصفح. يتكون من عدة كائنات أهمها window الذي يمثل نافذة المتصفح.

نوافذ التنبيه

// عرض تنبيه بسيط
alert('مرحباً بك!');

// طلب تأكيد من المستخدم
const confirmed = confirm('هل أنت متأكد؟');

// طلب إدخال من المستخدم
const name = prompt('ما هو اسمك؟', '');

المؤقتات

// تنفيذ كود بعد تأخير
setTimeout(() => {
  console.log('تم التنفيذ بعد 3 ثواني');
}, 3000);

// تنفيذ كود بشكل متكرر
const intervalId = setInterval(() => {
  console.log('يتم التنفيذ كل ثانيتين');
}, 2000);

// إيقاف التنفيذ المتكرر
clearInterval(intervalId);

Web Storage API

التخزين المحلي في المتصفح

توفر Web Storage API طريقتين لتخزين البيانات في متصفح المستخدم:

النوع المدة السعة نطاق الوصول
localStorage دائمة (حتى يتم حذفها) ~5-10MB نفس المصدر (origin)
sessionStorage لجلسة التصفح فقط ~5-10MB نفس التبويب

الوظائف الأساسية

// تخزين قيمة
localStorage.setItem('username', 'Ali');

// استرجاع قيمة
const username = localStorage.getItem('username');

// إزالة قيمة
localStorage.removeItem('username');

// مسح جميع القيم
localStorage.clear();

// تخزين كائن
const user = { name: 'Ali', age: 30 };
localStorage.setItem('user', JSON.stringify(user));

// استرجاع كائن
const storedUser = JSON.parse(localStorage.getItem('user'));

تجربة عملية

متى نستخدم localStorage؟

استخدم localStorage لتخزين التفضيلات، بيانات التهيئة، أو أي بيانات تريد استمرارها بين جلسات التصفح. استخدم sessionStorage للبيانات المؤقتة التي تحتاجها فقط خلال الجلسة الحالية.

JSON في JavaScript

ما هو JSON؟

JSON (JavaScript Object Notation) هو تنسيق خفيف الوزن لتبادل البيانات. وهو مكتوب بنص يمكن قراءته بواسطة الإنسان والآلة.

كائن JavaScript

const person = {
  name: "أحمد",
  age: 30,
  isStudent: false,
  courses: ["JavaScript", "React"],
  address: {
    city: "القاهرة",
    country: "مصر"
  }
};

JSON المقابل

{
  "name": "أحمد",
  "age": 30,
  "isStudent": false,
  "courses": ["JavaScript", "React"],
  "address": {
    "city": "القاهرة",
    "country": "مصر"
  }
}

JSON.stringify()

تحويل كائن JavaScript إلى سلسلة JSON

const user = {
  name: "سارة",
  email: "sara@example.com"
};

const jsonString = JSON.stringify(user);
console.log(jsonString);
// الناتج: {"name":"سارة","email":"sara@example.com"}

JSON.parse()

تحويل سلسلة JSON إلى كائن JavaScript

const jsonStr = '{"name":"سارة","email":"sara@example.com"}';

const userObj = JSON.parse(jsonStr);
console.log(userObj.name); // "سارة"
console.log(userObj.email); // "sara@example.com"

استخدام JSON مع localStorage

// تخزين كائن في localStorage
const settings = {
  theme: "dark",
  notifications: true,
  language: "ar"
};

localStorage.setItem('appSettings', JSON.stringify(settings));

// استرجاع الكائن من localStorage
const storedSettings = JSON.parse(localStorage.getItem('appSettings'));
console.log(storedSettings.theme); // "dark"

ملاحظة هامة

JSON لا يدعم جميع أنواع البيانات في JavaScript. على سبيل المثال، الدوال (Functions)، القيم غير المعرفة (undefined)، والرموز (Symbols) لا يمكن تحويلها إلى JSON.

البرمجة غير المتزامنة (Asynchronous Programming)

الفرق بين المتزامن وغير المتزامن

برمجة متزامنة

التنفيذ خطوة بخطوة، كل عملية تنتظر انتهاء التي قبلها

console.log("الخطوة 1");
console.log("الخطوة 2");
console.log("الخطوة 3");
// الناتج: 1, 2, 3

برمجة غير متزامنة

إمكانية تنفيذ عمليات متعددة في نفس الوقت دون الانتظار

console.log("البداية");

setTimeout(() => {
  console.log("عملية غير متزامنة");
}, 2000);

console.log("النهاية");
// الناتج: البداية, النهاية, عملية غير متزامنة

مثال على عملية غير متزامنة

لماذا نحتاج البرمجة غير المتزامنة؟

في تطبيقات الويب، العديد من العمليات تستغرق وقتًا (مثل جلب البيانات من خادم، قراءة ملفات، إلخ). البرمجة غير المتزامنة تمنع تجميد واجهة المستخدم أثناء انتظار هذه العمليات.

طابور الدوال الراجعة

اطلع علي الفيديو ده

عملية تنفيذ الكود

  1. يتم تحميل الكود وإنشاء سياق التنفيذ
  2. يتم وضع الدوال في Call Stack وتنفيذها
  3. العمليات غير المتزامنة تُرسل إلى Web APIs
  4. عند اكتمال العمليات، توضع الدوال الراجعة في Callback Queue
  5. Event Loop تنقل الدوال من Callback Queue إلى Call Stack عندما يكون فارغًا

مثال توضيحي

console.log("البداية");

setTimeout(() => {
  console.log("setTimeout");
}, 0);

Promise.resolve().then(() => {
  console.log("Promise");
});

console.log("النهاية");
// الناتج: البداية, النهاية, Promise, setTimeout

لماذا هذا الترتيب؟

Microtasks (مثل Promises) لها أولوية أعلى من Macrotasks (مثل setTimeout). لذلك يتم تنفيذها قبل Macrotasks عندما يكون Call Stack فارغًا.

// إضافة منتج إلى السلة
function addToCart(product) {
  let cart = JSON.parse(localStorage.getItem('cart')) || [];
  const existingItem = cart.find(item => item.id === product.id);

  if (existingItem) {
    existingItem.quantity += 1;
  } else {
    cart.push({ ...product, quantity: 1 });
  }

  localStorage.setItem('cart', JSON.stringify(cart));
  updateCartDisplay();
}

// تحديث عرض السلة
function updateCartDisplay() {
  const cart = JSON.parse(localStorage.getItem('cart')) || [];
  const container = document.getElementById('cart-items-container');
  container.innerHTML = '';
  let total = 0;

  if (cart.length === 0) {
    container.innerHTML = '<p class="text-center py-4">السلة فارغة</p>';
  } else {
    cart.forEach(item => {
      const itemTotal = item.price * item.quantity;
      total += itemTotal;

      const itemElement = document.createElement('div');
      itemElement.className = 'cart-item';
      itemElement.innerHTML = `
        <div class="flex items-center gap-3">
          <div class="w-10 h-10 bg-gray-200 rounded"></div>
          <div>
            <div>${item.name}</div>
            <div class="text-sm">${item.price} ج.م × ${item.quantity}</div>
          </div>
        </div>
        <div class="flex items-center gap-2">
          <span class="font-bold">${itemTotal} ج.م</span>
          <button class="text-red-500 remove-item" data-id="${item.id}"><i class="fas fa-trash"></i></button>
        </div>
      `;
      container.appendChild(itemElement);
    });
  }

  document.getElementById('cart-total').textContent = `${total} ج.م`;
}

المطلوب من الطالب

سنقوم بتحديث المشاريع السابقة التي أنشأتموها لتصبح أكثر تفاعلية باستخدام localStorage. إليك تفاصيل المهام:

تطبيق To-Do List

نظام تسجيل المستخدم

تعديل مشروع التخرج لاستخدام localStorage لتسجيل الدخول وتخزين المستخدمين

1. تطوير To-Do List

  1. افتح مشروع To-Do List الذي أنشأته سابقًا.
  2. احفظ المهام باستخدام localStorage.
  3. أعد تحميل المهام عند تشغيل الصفحة.
  4. أضف إمكانية حذف أو تعليم المهمة كمكتملة.

2. تطوير نظام تسجيل الدخول

  • استخدم مشروع التخرج (المكوّن من 3 صفحات: تسجيل، دخول، منتجات).
  • عند إنشاء حساب، خزّن المستخدم في localStorage.
  • بعد تسجيل الدخول، خزّن المستخدم الحالي في currentUser وانتقل إلى صفحة المنتجات.
  • إذا فشل تسجيل الدخول، اعرض رسالة تنبيه "الرجاء تسجيل بريد جديد".
