المصفوفات (Arrays)

ما هي المصفوفات؟

المصفوفات هي كائنات خاصة تخزن مجموعة من القيم في قائمة واحدة، تبدأ فهارسها من الصفر.

الوصول إلى العناصر

const fruits = ["تفاح", "موز", "برتقال"];

// الوصول باستخدام الفهرس
console.log(fruits[0]); // "تفاح"
console.log(fruits[2]); // "برتقال"

// عدد العناصر باستخدام length
console.log(fruits.length); // 3

الإضافة والحذف

let colors = ["أحمر", "أخضر"];

// إضافة عنصر في النهاية
colors.push("أزرق");

// إضافة عنصر في البداية
colors.unshift("أصفر");

// حذف آخر عنصر
colors.pop();

// حذف أول عنصر
colors.shift();

خصائص المصفوفات

  • المصفوفات قابلة للتغيير (mutable)
  • يمكن أن تحتوي على أنواع بيانات مختلفة
  • الفهارس تبدأ من الصفر (0-based indexing)
  • لها خاصية length لمعرفة عدد العناصر

  • معالجة عناصر قائمة مستخدمين
  • قراءة بيانات من مصفوفة أو قاعدة بيانات
  • إنشاء عناصر واجهة متكررة
  • معالجة صور متعددة في معرض

حلقة for

تُستخدم عندما تعرف عدد المرات التي تريد تكرار الكود فيها.

// بناء حلقة for
for (تعريف المتغير; الشرط; التحديث) {
  // الكود الذي سيتم تنفيذه
}
// مثال: طباعة الأرقام من 1 إلى 5
for (let i = 1; i <= 5; i++) {
  console.log(i); // 1, 2, 3, 4, 5
}

الفرق بين while و do...while

مثال عملي: لعبة التخمين باستخدام while
// الكمبيوتر يختار رقمًا عشوائيًا بين 1 و 100
const secretNumber = Math.floor(Math.random() * 100) + 1;
const maxTries = 5;
let attempts = 0;

alert("🎮 أهلاً بك في لعبة التخمين!\nاختر رقمًا بين 1 و 100. لديك 5 محاولات.");

while (attempts < maxTries) {
  const input = prompt(`🕹️ محاولة رقم ${attempts + 1}: أدخل رقمك`);
  const guess = Number(input);

  if (isNaN(guess) || guess < 1 || guess > 100) {
    alert("⚠️ أدخل رقمًا صحيحًا بين 1 و 100!");
    continue;
  }

  attempts++;

  if (guess === secretNumber) {
    alert(`🎉 مبروك! الرقم الصحيح هو ${secretNumber}`);
    break;
  } else if (guess < secretNumber) {
    alert("📈 الرقم أكبر من تخمينك.");
  } else {
    alert("📉 الرقم أصغر من تخمينك.");
  }
}

alert(`❌ انتهت المحاولات! الرقم الصحيح كان: ${secretNumber}`);

🧠 الميزة: نستخدم while لأن عدد المحاولات معلوم مسبقًا ولكن نتيجة كل محاولة غير معلومة، والشرط يتم التحقق منه في كل مرة.

مثال: تسجيل الدخول
// باستخدام do...while
let password;

do {
  password = prompt("أدخل كلمة السر:");
} while (password !== "1234");

alert("مرحبًا بك!");

✅ الميزة: التنفيذ الأولي قبل التحقق

مثال: قائمة الخيارات
let choice;

do {
  choice = prompt("اختر: 1-وقت 2-رسالة 3-خروج");
  // معالجة الاختيار
} while (choice !== "3");

alert("تم الخروج!");

✅ الميزة: عرض القائمة مرة واحدة على الأقل

المقارنة while do...while
التحقق من الشرط قبل التنفيذ ✅ نعم ❌ لا
تنفيذ الكود مرة واحدة على الأقل ❌ لا ✅ نعم
الاستخدام الأمثل عندما لا تريد تنفيذ الكود إطلاقًا إذا لم يتحقق الشرط عندما تريد تنفيذ الكود مرة واحدة على الأقل قبل التحقق
مثال من الواقع قراءة ملف حتى نهايته نظام القوائم، تسجيل الدخول

الحل الأنسب: do...while
  • ننفذ الكود مرة واحدة على الأقل
  • لا حاجة لتهيئة غريبة للمتغير
  • الكود أوضح وأسهل للقراءة

do {
  password = prompt("أدخل كلمة السر:");

alert("مرحبًا بك!");
لماذا هو الأفضل؟
نحتاج تنفيذ الكود مرة واحدة على الأقل قبل التحقق من الشرط.
المقارنة while do...while
يتحقق من الشرط قبل التنفيذ؟ ✅ نعم ❌ لا
ينفذ مرة واحدة على الأقل؟ ❌ لا ✅ نعم
الأفضل عندما تحتاج تنفيذ أولي قبل التحقق ❌ ✅
القاعدة الذهبية: إذا كنت بحاجة إلى تنفيذ الكود على الأقل مرة واحدة قبل التحقق من الشرط، استخدم do...while.

الدوال والمعاملات (Functions & Parameters)

ما هي الدوال؟

الدوال هي كتل من التعليمات البرمجية يمكن تنفيذها عند استدعائها. تساعد في تنظيم الكود وإعادة استخدامه.

5
مدخل
ƒ(x)
25
مخرج

تعريف الدوال

// Function Declaration
function greet(name) {
  return "مرحبًا " + name;
}

// Function Expression
const greet = function(name) {
  return "مرحبًا " + name;
};

// Arrow Function
const greet = name => "مرحبًا " + name;

المعاملات المتقدمة

// المعاملات الافتراضية
function greet(name = "زائر") {
  return "مرحبًا " + name;
}

// rest parameters
function sum(...numbers) {
  return numbers.reduce((a, b) => a + b);
}

تحدي الدوال

أنشئ دالة تقبل أي عدد من الأرقام وتعيد حاصل ضربها. استخدم rest parameters.

متوسط

طرق تعريف الدوال

Function Declaration

تعريف تقليدي للدالة يمكن استدعاؤه قبل تعريفه

// تعريف الدالة
function greet(name) {
  return `مرحبًا ${name}!`;
}

// استدعاء الدالة
console.log(greet("أحمد")); // "مرحبًا أحمد!"

Function Expression

تعيين دالة لمتغير (لا يمكن استدعاؤها قبل التعريف)

// تعريف الدالة
const greet = function(name) {
  return `مرحبًا ${name}!`;
};

// استدعاء الدالة
console.log(greet("سارة")); // "مرحبًا سارة!"

الفرق بين الطريقتين

  • الدوال المعرفة بالإعلان (Declaration) يمكن استدعاؤها قبل تعريفها
  • الدوال المعرفة بالتعبير (Expression) لا يمكن استدعاؤها قبل تعريفها
  • الدوال المعرفة بالتعبير أكثر مرونة في الاستخدام

الدالة الوصف مثال النتيجة
forEach() تنفيذ دالة لكل عنصر [1,2,3].forEach(n => console.log(n)) 1, 2, 3 في الكونسول
